        <div id="gameArea">
            <div id="player"></div>
        </div>
    </div>
    <script>
        const player = document.getElementById('player');
        const gameArea = document.getElementById('gameArea');
        const gravity = 0.5;
        const jumpStrength = 10;
        const speed = 5;
        let posX = 225;
        let posY = 0;
        let velocityY = 0;
        let onGround = false;

        function updatePlayerPosition() {
            velocityY += gravity;
            posY += velocityY;

            // Check for collision with the ground
            if (posY + player.offsetHeight > gameArea.offsetHeight) {
                posY = gameArea.offsetHeight - player.offsetHeight;
                velocityY = 0;
                onGround = true;
            }

            player.style.left = posX + 'px';
            player.style.top = posY + 'px';
        }

        function movePlayer(e) {
            if (e.key === 'ArrowLeft') {
                posX -= speed;
            } else if (e.key === 'ArrowRight') {
                posX += speed;
            } else if (e.key === ' ' && onGround) {
                velocityY = -jumpStrength;
                onGround = false;
            }

            // Ensure the player stays within the game area horizontally
            if (posX < 0) {
                posX = 0;
            } else if (posX + player.offsetWidth > gameArea.offsetWidth) {
                posX = gameArea.offsetWidth - player.offsetWidth;
            }
        }

        document.addEventListener('keydown', movePlayer);
        
        function gameLoop() {
            updatePlayerPosition();
            requestAnimationFrame(gameLoop);
        }

        gameLoop();
    </script>
